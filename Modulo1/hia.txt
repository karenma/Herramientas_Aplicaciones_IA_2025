    #Método para obtener las columnas y dataframe binarios
    def get_binarios(self):
        col_bin = []
        for col in self.data.columns:
            if self.data[col].nunique() == 2:
                col_bin.append(col)
        return self.data[col_bin], col_bin

    #remover signos de puntuacion 
    @staticmethod
    def remove_punctuation(x):
        try:
            x = ''.join(ch for ch in x if ch not in string.punctuation)
        except:
            print(f'{x} no es una cadena de caracteres')
            pass
        return x

    def remove_digits(x):
        try:
            x=''.join(ch for ch in x if ch not in string.digits)
        except:
            print(f'{x} no es una cadena de caracteres')
            pass
        return x
    # remover espacios en blanco
    def remove_whitespace(x):
        try:
            x=' '.join(x.split())
        except:
            pass
        return x

    def dqr(df):
    # Lista de nombres de columnas
    columns = pd.DataFrame(df.columns, columns=['Column_Name'], index=df.columns)

    # Tipos de datos
    data_dtypes = pd.DataFrame(df.dtypes, columns=['Dtypes'])

    # Valores presentes
    present_values = pd.DataFrame(df.count(), columns=['Present_values'])

    # Valores faltantes
    missing_values = pd.DataFrame(df.isnull().sum(), columns=['Missing_values'])

    # Valores únicos
    unique_values = pd.DataFrame(df.nunique(), columns=['Unique_values'])

    # ---------------------------------------------------
    # 1. Columna booleana Is_categorical
    # ---------------------------------------------------
    is_categorical = pd.DataFrame(
        df.dtypes.apply(lambda x: x == "object" or str(x) == "category"),
        columns=["Is_categorical"]
    )

    # ---------------------------------------------------
    # 2. Categorías solo para columnas categóricas
    # ---------------------------------------------------
    categories = pd.DataFrame(columns=["Categories"])

    for col in df.columns:
        if is_categorical.loc[col, "Is_categorical"]:
            if df[col].nunique() <= 10:
                categories.loc[col] = [list(df[col].dropna().unique())]
            else:
                categories.loc[col] = ["Más de 10 categorías"]
        else:
            categories.loc[col] = ["N/A"]

    # ---------------------------------------------------
    # 3. Estadísticas solo para columnas numéricas
    # ---------------------------------------------------
    max_values = pd.DataFrame(columns=['Max'])
    min_values = pd.DataFrame(columns=['Min'])
    mean_values = pd.DataFrame(columns=['Mean'])
    std_values = pd.DataFrame(columns=['Std'])

    for col in df.columns:
        if not is_categorical.loc[col, "Is_categorical"]:
            try:
                max_values.loc[col] = [df[col].max()]
                min_values.loc[col] = [df[col].min()]
                mean_values.loc[col] = [df[col].mean()]
                std_values.loc[col] = [df[col].std()]
            except:
                max_values.loc[col] = ["N/A"]
                min_values.loc[col] = ["N/A"]
                mean_values.loc[col] = ["N/A"]
                std_values.loc[col] = ["N/A"]
        else:
            max_values.loc[col] = ["N/A"]
            min_values.loc[col] = ["N/A"]
            mean_values.loc[col] = ["N/A"]
            std_values.loc[col] = ["N/A"]

    # ---------------------------------------------------
    # Ensamblar el reporte final
    # ---------------------------------------------------
    dqr_report = (columns
                  .join(data_dtypes)
                  .join(present_values)
                  .join(missing_values)
                  .join(unique_values)
                  .join(is_categorical)
                  .join(categories)
                  .join(max_values)
                  .join(min_values)
                  .join(mean_values)
                  .join(std_values)
                 )

    return dqr_report
